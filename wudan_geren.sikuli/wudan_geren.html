def loop_check(img_target, times = 20, raise_error = True):
    max_retry = times
    img_start = exists(img_target)
    while not img_start:
        wait(10)
        img_start = exists(img_target)
        max_retry = max_retry - 1
        if max_retry == 0:
            if raise_error:
                raise Exception('image target should have been found')
            else:
                return 0
    click(img_start)

def ti_jiao():
